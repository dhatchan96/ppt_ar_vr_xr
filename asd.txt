class JiraTicketsView:
    def processSelectedValues(self, selected_values_json):
        """
        Process the selected values sent from JavaScript.
        :param selected_values_json: A JSON string of selected values.
        """
        import json

        # Parse the JSON string into a Python list
        selected_values = json.loads(selected_values_json)

        print(f"Received selected values: {selected_values}")
        # Perform your desired processing here
        return f"Processed {len(selected_values)} values: {selected_values}"
