<script>
    // Replace the default alert function with a custom implementation
    window.alert = function(message) {
        // Create a modal container dynamically
        const modal = document.createElement('div');
        modal.style.position = 'fixed';
        modal.style.top = '50%';
        modal.style.left = '50%';
        modal.style.transform = 'translate(-50%, -50%)';
        modal.style.backgroundColor = '#fff';
        modal.style.padding = '20px';
        modal.style.boxShadow = '0px 4px 6px rgba(0, 0, 0, 0.1)';
        modal.style.borderRadius = '8px';
        modal.style.zIndex = '10000';
        modal.style.textAlign = 'center';

        // Add the message to the modal
        modal.innerHTML = `
            <p style="margin-bottom: 20px; font-size: 16px; font-family: Arial, sans-serif;">
                ${message}
            </p>
            <button onclick="document.body.removeChild(this.parentElement)" style="
                padding: 8px 16px;
                background-color: #007bff;
                color: white;
                border: none;
                border-radius: 4px;
                cursor: pointer;
                font-size: 14px;
            ">
                OK
            </button>
        `;

        // Append the modal to the document body
        document.body.appendChild(modal);
    };
</script>
