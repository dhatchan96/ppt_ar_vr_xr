import json
from urllib.parse import unquote

def process_data(self, data):
    """
    Process the data received from JavaScript.
    :param data: The encoded data from the URL.
    """
    # Decode the URL-encoded data
    decoded_data = unquote(data)
    print(f"Decoded data: {decoded_data}")
    
    # Convert the decoded string into a Python list
    try:
        # If it's a JSON string, load it
        values = json.loads(decoded_data)
    except json.JSONDecodeError:
        # If it's not JSON, split it into a list by commas
        values = decoded_data.strip("[]").replace('"', '').split(", ")
    
    print(f"Received values as list: {values}")
