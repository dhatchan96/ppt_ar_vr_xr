class CustomHyperBrowser(HyperBrowser):
    def __init__(self, html_content, python_object, **kwargs):
        """
        Extend the HyperBrowser to include a scriptable Python object.
        :param html_content: The HTML content to render.
        :param python_object: The Python object to expose to JavaScript.
        """
        super().__init__(src=html_content, **kwargs)
        self.python_object = python_object  # Store the Python object

    def get_scriptable_object(self):
        """
        Return the Python object to expose as window.external.
        """
        return self.python_object
