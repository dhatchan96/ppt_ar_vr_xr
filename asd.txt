
@dag.cellfn
def TemplateParams(self):
    build = self.Build()
    regressionJira = build.externalSystemIds.get('regressionJiraTicket', '')
    jiraTable = self.JiraTicketsTable()
    jira_tickets = {} if jiraTable is dag.NO_VALUE else jiraTable

    # Fetch linked JIRAs for each ticket
    for ticket in jira_tickets:
        ticket_id = ticket.get("Ticket")  # Adjust based on your ticket field name
        ticket["linked_jiras"] = self.fetch_linked_jiras(ticket_id)  # Fetch linked JIRAs

    jira_changes = {} if jiraTable is dag.NO_VALUE else self.JiraChanges()
    popup_message = self.Popup()

    return dict(
        build=build,
        jira_tickets=jira_tickets,
        jira_changes=jira_changes,
        regressionJira=regressionJira,
        popup_message=popup_message
    )
