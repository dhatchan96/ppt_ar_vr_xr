from jira import JIRA

def fetch_linked_jiras(self, ticket_id):
    """
    Fetch inward linked JIRA tickets for the given ticket ID.

    Args:
        ticket_id (str): The JIRA ticket ID to fetch linked issues for.

    Returns:
        list: A list of linked JIRA ticket IDs (inward links).
    """
    try:
        # Get the issue object for the given ticket
        issue = self.jira_client.issue(ticket_id)  # Assuming self.jira_client is a valid JIRA client instance

        linked_jiras = []

        # Iterate through the issue links
        for link in issue.fields.issuelinks:
            if hasattr(link, "inwardIssue"):  # Check if the link has an inward issue
                linked_jiras.append(link.inwardIssue.key)  # Append the key of the inward linked issue

        return linked_jiras

    except Exception as e:
        print(f"Error fetching linked JIRAs for ticket {ticket_id}: {e}")
        return []
