<script>
    function showPopup(message) {
        if (message) {
            alert(message); // Show a simple popup
        }
    }

    // Automatically trigger the popup if a message exists
    document.addEventListener("DOMContentLoaded", function() {
        const popupMessage = "{{ popup_message }}"; // Injected by Python
        if (popupMessage) {
            showPopup(popupMessage);
        }
    });
</script>
