<script>
    function sendSelectedValuesToPython() {
        const checkboxes = document.querySelectorAll('input[name="jiraSelect"]:checked');
        const selectedValues = Array.from(checkboxes).map(cb => cb.value);

        if (selectedValues.length > 0) {
            const data = encodeURIComponent(JSON.stringify(selectedValues));
            // Use a custom protocol to send data to Python
            window.location.href = `custom-protocol:${data}`;
        } else {
            alert("No rows selected.");
        }
    }
</script>

<button onclick="sendSelectedValuesToPython()">Send to Python</button>
