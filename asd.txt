class JiraTicketsView:
    def handle_navigation(self, event):
        """
        Handle navigation events from the HyperBrowser.
        :param event: The navigation event object.
        """
        print(f"Navigation URL intercepted: {event.url}")
        if event.url.startswith("custom-protocol:"):
            # Extract the data after the protocol
            data = event.url[len("custom-protocol:"):]
            print(f"Custom protocol data: {data}")

            # Call a method to process the data
            self.process_data(data)
            return True  # Prevent default navigation

        return False  # Allow default navigation

    def process_data(self, data):
        """
        Process the data received from JavaScript.
        :param data: The encoded data from JavaScript.
        """
        import json
        decoded_data = json.loads(data)  # Decode the JSON data
        print(f"Data received in Python: {decoded_data}")
