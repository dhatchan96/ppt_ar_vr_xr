<script>
    function display() {
        const checkboxes = document.querySelectorAll('input[name="jiraSelect"]:checked');
        const selectedValues = [];

        for (let i = 0; i < checkboxes.length; i++) {
            selectedValues.push(checkboxes[i].value);
        }

        if (selectedValues.length > 0) {
            alert("Selected Rows: " + selectedValues.join(", "));

            // Pass the selected values to the Python method
            if (window.external && window.external.processSelectedValues) {
                window.external.processSelectedValues(JSON.stringify(selectedValues));
            } else {
                console.error("Python bridge is not available.");
            }
        } else {
            alert("No rows selected.");
        }
    }
</script>
